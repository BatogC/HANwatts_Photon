1 /**
2  * MFRC522.h - Library to use ARDUINO RFID MODULE KIT 13.56 MHZ WITH TAGS SPI W AND R BY COOQROBOT.
3  * Based on code Dr.Leong ( WWW.B2CQSHOP.COM )
4  * Created by Miguel Balboa (circuitito.com), Jan, 2012.
5  * Rewritten by Søren Thing Andersen (access.thing.dk), fall of 2013 (Translation to English, refactored, comments, anti collision, cascade levels.)
6  * Released into the public domain.
7  *
8  * Please read this file for an overview and then MFRC522.cpp for comments on the specific functions.
9  * Search for "mf-rc522" on ebay.com to purchase the MF-RC522 board.
10  *
11  * There are three hardware components involved:
12  * 1) The micro controller: An Arduino
13  * 2) The PCD (short for Proximity Coupling Device): NXP MFRC522 Contactless Reader IC
14  * 3) The PICC (short for Proximity Integrated Circuit Card): A card or tag using the ISO 14443A interface, eg Mifare or NTAG203.
15  *
16  * The microcontroller and card reader uses SPI for communication.
17  * The protocol is described in the MFRC522 datasheet: http://www.nxp.com/documents/data_sheet/MFRC522.pdf
18  *
19  * The card reader and the tags communicate using a 13.56MHz electromagnetic field.
20  * The protocol is defined in ISO/IEC 14443-3 Identification cards -- Contactless integrated circuit cards -- Proximity cards -- Part 3: Initialization and anticollision".
21  * A free version of the final draft can be found at http://wg8.de/wg8n1496_17n3613_Ballot_FCD14443-3.pdf
22  * Details are found in chapter 6, Type A – Initialization and anticollision.
23  *
24  * If only the PICC UID is wanted, the above documents has all the needed information.
25  * To read and write from MIFARE PICCs, the MIFARE protocol is used after the PICC has been selected.
26  * The MIFARE Classic chips and protocol is described in the datasheets:
27  * 1K: http://www.nxp.com/documents/data_sheet/MF1S503x.pdf
28  * 4K: http://www.nxp.com/documents/data_sheet/MF1S703x.pdf
29  * Mini: http://www.idcardmarket.com/download/mifare_S20_datasheet.pdf
30  * The MIFARE Ultralight chip and protocol is described in the datasheets:
31  * Ultralight: http://www.nxp.com/documents/data_sheet/MF0ICU1.pdf
32  * Ultralight C: http://www.nxp.com/documents/short_data_sheet/MF0ICU2_SDS.pdf
33  *
34  * MIFARE Classic 1K (MF1S503x):
35  * Has 16 sectors * 4 blocks/sector * 16 bytes/block = 1024 bytes.
36  * The blocks are numbered 0-63.
37  * Block 3 in each sector is the Sector Trailer. See http://www.nxp.com/documents/data_sheet/MF1S503x.pdf sections 8.6 and 8.7:
38  * Bytes 0-5: Key A
39  * Bytes 6-8: Access Bits
40  * Bytes 9: User data
41  * Bytes 10-15: Key B (or user data)
42  * Block 0 is read only manufacturer data.
43  * To access a block, an authentication using a key from the block's sector must be performed first.
44  * Example: To read from block 10, first authenticate using a key from sector 3 (blocks 8-11).
45  * All keys are set to FFFFFFFFFFFFh at chip delivery.
46  * Warning: Please read section 8.7 "Memory Access". It includes this text: if the PICC detects a format violation the whole sector is irreversibly blocked.
47  * To use a block in "value block" mode (for Increment/Decrement operations) you need to change the sector trailer. Use PICC_SetAccessBits() to calculate the bit patterns.
48  * MIFARE Classic 4K (MF1S703x):
49  * Has (32 sectors * 4 blocks/sector + 8 sectors * 16 blocks/sector) * 16 bytes/block = 4096 bytes.
50  * The blocks are numbered 0-255.
51  * The last block in each sector is the Sector Trailer like above.
52  * MIFARE Classic Mini (MF1 IC S20):
53  * Has 5 sectors * 4 blocks/sector * 16 bytes/block = 320 bytes.
54  * The blocks are numbered 0-19.
55  * The last block in each sector is the Sector Trailer like above.
56  *
57  * MIFARE Ultralight (MF0ICU1):
58  * Has 16 pages of 4 bytes = 64 bytes.
59  * Pages 0 + 1 is used for the 7-byte UID.
60  * Page 2 contains the last chech digit for the UID, one byte manufacturer internal data, and the lock bytes (see http://www.nxp.com/documents/data_sheet/MF0ICU1.pdf section 8.5.2)
61  * Page 3 is OTP, One Time Programmable bits. Once set to 1 they cannot revert to 0.
62  * Pages 4-15 are read/write unless blocked by the lock bytes in page 2.
63  * MIFARE Ultralight C (MF0ICU2):
64  * Has 48 pages of 4 bytes = 64 bytes.
65  * Pages 0 + 1 is used for the 7-byte UID.
66  * Page 2 contains the last chech digit for the UID, one byte manufacturer internal data, and the lock bytes (see http://www.nxp.com/documents/data_sheet/MF0ICU1.pdf section 8.5.2)
67  * Page 3 is OTP, One Time Programmable bits. Once set to 1 they cannot revert to 0.
68  * Pages 4-39 are read/write unless blocked by the lock bytes in page 2.
69  * Page 40 Lock bytes
70  * Page 41 16 bit one way counter
71  * Pages 42-43 Authentication configuration
72  * Pages 44-47 Authentication key
73  */

194  // Commands sent to the PICC.
195  enum PICC_Command {
196  // The commands used by the PCD to manage communication with several PICCs (ISO 14443-3, Type A, section 6.4)
197  PICC_CMD_REQA = 0x26, // REQuest command, Type A. Invites PICCs in state IDLE to go to READY and prepare for anticollision or selection. 7 bit frame.
198  PICC_CMD_WUPA = 0x52, // Wake-UP command, Type A. Invites PICCs in state IDLE and HALT to go to READY(*) and prepare for anticollision or selection. 7 bit frame.
199  PICC_CMD_CT = 0x88, // Cascade Tag. Not really a command, but used during anti collision.
200  PICC_CMD_SEL_CL1 = 0x93, // Anti collision/Select, Cascade Level 1
201  PICC_CMD_SEL_CL2 = 0x95, // Anti collision/Select, Cascade Level 1
202  PICC_CMD_SEL_CL3 = 0x97, // Anti collision/Select, Cascade Level 1
203  PICC_CMD_HLTA = 0x50, // HaLT command, Type A. Instructs an ACTIVE PICC to go to state HALT.
204  // The commands used for MIFARE Classic (from http://www.nxp.com/documents/data_sheet/MF1S503x.pdf, Section 9)
205  // Use PCD_MFAuthent to authenticate access to a sector, then use these commands to read/write/modify the blocks on the sector.
206  // The read/write commands can also be used for MIFARE Ultralight.
207  PICC_CMD_MF_AUTH_KEY_A = 0x60, // Perform authentication with Key A
208  PICC_CMD_MF_AUTH_KEY_B = 0x61, // Perform authentication with Key B
209  PICC_CMD_MF_READ = 0x30, // Reads one 16 byte block from the authenticated sector of the PICC. Also used for MIFARE Ultralight.
210  PICC_CMD_MF_WRITE = 0xA0, // Writes one 16 byte block to the authenticated sector of the PICC. Called "COMPATIBILITY WRITE" for MIFARE Ultralight.
211  PICC_CMD_MF_DECREMENT = 0xC0, // Decrements the contents of a block and stores the result in the internal data register.
212  PICC_CMD_MF_INCREMENT = 0xC1, // Increments the contents of a block and stores the result in the internal data register.
213  PICC_CMD_MF_RESTORE = 0xC2, // Reads the contents of a block into the internal data register.
214  PICC_CMD_MF_TRANSFER = 0xB0, // Writes the contents of the internal data register to a block.
215  // The commands used for MIFARE Ultralight (from http://www.nxp.com/documents/data_sheet/MF0ICU1.pdf, Section 8.6)
216  // The PICC_CMD_MF_READ and PICC_CMD_MF_WRITE can also be used for MIFARE Ultralight.
217  PICC_CMD_UL_WRITE = 0xA2 // Writes one 4 byte page to the PICC.
218  };

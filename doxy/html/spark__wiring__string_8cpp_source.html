39 void dtoa (double val, unsigned char prec, char *sout) {
40  bool negative = val<0;
41  if (negative) {
42  val = -val;
43  *sout++ = '-';
44  }
45  long scale = 1;
46  for (uint8_t i=0; i<prec; i++)
47  scale *= 10;
48  val *= scale; // capture all the significant digits
49  uint64_t fixed = uint64_t(val);
50  if ((val-fixed)>=0.5) // round last digit
51  fixed++;
52 
53  unsigned long first = (unsigned long)(fixed / scale);
54  unsigned long second = (unsigned long)(fixed % scale);
55 
56  ultoa(first, sout, 10, 1);
57  if (prec) {
58  sout += strlen(sout);
59  *sout++ = '.';
60  ultoa(second, sout, 10, prec);
61  }
62 }

647 String& String::replace(const String& find, const String& replace)
648 {
649  if (len == 0 || find.len == 0) return *this;
650  int diff = replace.len - find.len;
651  char *readFrom = buffer;
652  char *foundAt;
653  if (diff == 0) {
654  while ((foundAt = strstr(readFrom, find.buffer)) != NULL) {
655  memcpy(foundAt, replace.buffer, replace.len);
656  readFrom = foundAt + replace.len;
657  }
658  } else if (diff < 0) {
659  char *writeTo = buffer;
660  while ((foundAt = strstr(readFrom, find.buffer)) != NULL) {
661  unsigned int n = foundAt - readFrom;
662  memcpy(writeTo, readFrom, n);
663  writeTo += n;
664  memcpy(writeTo, replace.buffer, replace.len);
665  writeTo += replace.len;
666  readFrom = foundAt + find.len;
667  len += diff;
668  }
669  strcpy(writeTo, readFrom);
670  } else {
671  unsigned int size = len; // compute size needed for result
672  while ((foundAt = strstr(readFrom, find.buffer)) != NULL) {
673  readFrom = foundAt + find.len;
674  size += diff;
675  }
676  if (size == len) return *this;;
677  if (size > capacity && !changeBuffer(size)) return *this; // XXX: tell user!
678  int index = len - 1;
679  while (index >= 0 && (index = lastIndexOf(find, index)) >= 0) {
680  readFrom = buffer + index + find.len;
681  memmove(readFrom + diff, readFrom, len - (readFrom - buffer));
682  len += diff;
683  buffer[len] = 0;
684  memcpy(buffer + index, replace.buffer, replace.len);
685  index--;
686  }
687  }

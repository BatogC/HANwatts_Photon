93 /**
94  * Reads a number of bytes from the specified register in the MFRC522 chip.
95  * The interface is described in the datasheet section 8.1.2.
96  */
97 void MFRC522::PCD_ReadRegister( byte reg, ///< The register to read from. One of the PCD_Register enums.
98  byte count, ///< The number of bytes to read
99  byte *values, ///< Byte array to store the values in.
100  byte rxAlign ///< Only bit positions rxAlign..7 in values[0] are updated.
101  ) {
102  if (count == 0) {
103  return;
104  }
105  //Serial.print("Reading "); Serial.print(count); Serial.println(" bytes from register.");
106  byte address = 0x80 | (reg & 0x7E); // MSB == 1 is for reading. LSB is not used in address. Datasheet section 8.1.2.3.
107  byte index = 0; // Index in values array.
108  digitalWrite(_chipSelectPin, LOW); // Select slave
109  count--; // One read is performed outside of the loop
110  SPI.transfer(address); // Tell MFRC522 which address we want to read
111  while (index < count) {
112  if (index == 0 && rxAlign) { // Only update bit positions rxAlign..7 in values[0]
113  // Create bit mask for bit positions rxAlign..7
114  byte mask = 0;
115  for (byte i = rxAlign; i <= 7; i++) {
116  mask |= (1 << i);
117  }
118  // Read value and tell that we want to read the same address again.
119  byte value = SPI.transfer(address);
120  // Apply mask to both current value of values[0] and the new data in value.
121  values[0] = (values[index] & ~mask) | (value & mask);
122  }
123  else { // Normal case
124  values[index] = SPI.transfer(address); // Read value and tell that we want to read the same address again.
125  }
126  index++;
127  }
128  values[index] = SPI.transfer(0); // Read the final byte. Send 0 to stop reading.
129  digitalWrite(_chipSelectPin, HIGH); // Release slave again
130 } // End PCD_ReadRegister()

161  byte length, ///< In: The number of bytes to transfer.
162  byte *result ///< Out: Pointer to result buffer. Result is written to result[0..1], low byte first.
163  ) {

165  PCD_WriteRegister(DivIrqReg, 0x04); // Clear the CRCIRq interrupt request bit
166  PCD_SetRegisterBitMask(FIFOLevelReg, 0x80); // FlushBuffer = 1, FIFO initialization
167  PCD_WriteRegister(FIFODataReg, length, data); // Write data to the FIFO
168  PCD_WriteRegister(CommandReg, PCD_CalcCRC); // Start the calculation

171  word i = 5000;
172  byte n;
173  while (1) {
174  n = PCD_ReadRegister(DivIrqReg); // DivIrqReg[7..0] bits are: Set2 reserved reserved MfinActIRq reserved CRCIRq reserved reserved
175  if (n & 0x04) { // CRCIRq bit set - calculation done

196  * Initializes the MFRC522 chip.

198 void MFRC522::PCD_Init() {
199  if (digitalRead(_resetPowerDownPin) == LOW) { //The MFRC522 chip is in power down mode.
200  digitalWrite(_resetPowerDownPin, HIGH); // Exit power down mode. This triggers a hard reset.
201  // Section 8.8.2 in the datasheet says the oscillator start-up time is the start up time of the crystal + 37,74�s. Let us be generous: 50ms.
202  delay(50);
203  }
204  else { // Perform a soft reset
205  PCD_Reset();
206  }
207 
208  // When communicating with a PICC we need a timeout if something goes wrong.
209  // f_timer = 13.56 MHz / (2*TPreScaler+1) where TPreScaler = [TPrescaler_Hi:TPrescaler_Lo].
210  // TPrescaler_Hi are the four low bits in TModeReg. TPrescaler_Lo is TPrescalerReg.
211  PCD_WriteRegister(TModeReg, 0x80); // TAuto=1; timer starts automatically at the end of the transmission in all communication modes at all speeds
212  PCD_WriteRegister(TPrescalerReg, 0xA9); // TPreScaler = TModeReg[3..0]:TPrescalerReg, ie 0x0A9 = 169 => f_timer=40kHz, ie a timer period of 25�s.
213  PCD_WriteRegister(TReloadRegH, 0x03); // Reload timer with 0x3E8 = 1000, ie 25ms before timeout.
214  PCD_WriteRegister(TReloadRegL, 0xE8);
215 
216  PCD_WriteRegister(TxASKReg, 0x40); // Default 0x00. Force a 100 % ASK modulation independent of the ModGsPReg register setting
217  PCD_WriteRegister(ModeReg, 0x3D); // Default 0x3F. Set the preset value for the CRC coprocessor for the CalcCRC command to 0x6363 (ISO 14443-3 part 6.2.4)
218  PCD_AntennaOn(); // Enable the antenna driver pins TX1 and TX2 (they were disabled by the reset)
219 } // End PCD_Init()

299 /**
300  * Transfers data to the MFRC522 FIFO, executes a commend, waits for completion and transfers data back from the FIFO.
301  * CRC validation can only be done if backData and backLen are specified.
302  *
303  * @return STATUS_OK on success, STATUS_??? otherwise.
304  */
305 byte MFRC522::PCD_CommunicateWithPICC( byte command, ///< The command to execute. One of the PCD_Command enums.
306  byte waitIRq, ///< The bits in the ComIrqReg register that signals successful completion of the command.
307  byte *sendData, ///< Pointer to the data to transfer to the FIFO.
308  byte sendLen, ///< Number of bytes to transfer to the FIFO.
309  byte *backData, ///< NULL or pointer to buffer if data should be read back after executing the command.
310  byte *backLen, ///< In: Max number of bytes to write to *backData. Out: The number of bytes returned.
311  byte *validBits, ///< In/Out: The number of valid bits in the last byte. 0 for 8 valid bits.
312  byte rxAlign, ///< In: Defines the bit position in backData[0] for the first bit received. Default 0.
313  bool checkCRC ///< In: True => The last two bytes of the response is assumed to be a CRC_A that must be validated.
314  ) {
315  byte n, _validBits;
316  unsigned int i;
317 
318  // Prepare values for BitFramingReg
319  byte txLastBits = validBits ? *validBits : 0;
320  byte bitFraming = (rxAlign << 4) + txLastBits; // RxAlign = BitFramingReg[6..4]. TxLastBits = BitFramingReg[2..0]
321 
322  PCD_WriteRegister(CommandReg, PCD_Idle); // Stop any active command.
323  PCD_WriteRegister(ComIrqReg, 0x7F); // Clear all seven interrupt request bits
324  PCD_SetRegisterBitMask(FIFOLevelReg, 0x80); // FlushBuffer = 1, FIFO initialization
325  PCD_WriteRegister(FIFODataReg, sendLen, sendData); // Write sendData to the FIFO
326  PCD_WriteRegister(BitFramingReg, bitFraming); // Bit adjustments
327  PCD_WriteRegister(CommandReg, command); // Execute the command
328  if (command == PCD_Transceive) {
329  PCD_SetRegisterBitMask(BitFramingReg, 0x80); // StartSend=1, transmission of data starts
330  }
331 
332  // Wait for the command to complete.
333  // In PCD_Init() we set the TAuto flag in TModeReg. This means the timer automatically starts when the PCD stops transmitting.
334  // Each iteration of the do-while-loop takes 17.86�s.
335  i = 2000;
336  while (1) {
337  n = PCD_ReadRegister(ComIrqReg); // ComIrqReg[7..0] bits are: Set1 TxIRq RxIRq IdleIRq HiAlertIRq LoAlertIRq ErrIRq TimerIRq
338  if (n & waitIRq) { // One of the interrupts that signal success has been set.
339  break;
340  }
341  if (n & 0x01) { // Timer interrupt - nothing received in 25ms
342  return STATUS_TIMEOUT;
343  }
344  if (--i == 0) { // The emergency break. If all other condions fail we will eventually terminate on this one after 35.7ms. Communication with the MFRC522 might be down.
345  return STATUS_TIMEOUT;
346  }
347  }
348 
349  // Stop now if any errors except collisions were detected.
350  byte errorRegValue = PCD_ReadRegister(ErrorReg); // ErrorReg[7..0] bits are: WrErr TempErr reserved BufferOvfl CollErr CRCErr ParityErr ProtocolErr
351  if (errorRegValue & 0x13) { // BufferOvfl ParityErr ProtocolErr
352  return STATUS_ERROR;
353  }
354 
355  // If the caller wants data back, get it from the MFRC522.
356  if (backData && backLen) {
357  n = PCD_ReadRegister(FIFOLevelReg); // Number of bytes in the FIFO
358  if (n > *backLen) {
359  return STATUS_NO_ROOM;
360  }
361  *backLen = n; // Number of bytes returned
362  PCD_ReadRegister(FIFODataReg, n, backData, rxAlign); // Get received data from FIFO
363  _validBits = PCD_ReadRegister(ControlReg) & 0x07; // RxLastBits[2:0] indicates the number of valid bits in the last received byte. If this value is 000b, the whole byte is valid.
364  if (validBits) {
365  *validBits = _validBits;
366  }
367  }
368 
369  // Tell about collisions
370  if (errorRegValue & 0x08) { // CollErr
371  return STATUS_COLLISION;
372  }
373 
374  // Perform CRC_A validation if requested.
375  if (backData && backLen && checkCRC) {
376  // In this case a MIFARE Classic NAK is not OK.
377  if (*backLen == 1 && _validBits == 4) {
378  return STATUS_MIFARE_NACK;
379  }
380  // We need at least the CRC_A value and all 8 bits of the last byte must be received.
381  if (*backLen < 2 || _validBits != 0) {
382  return STATUS_CRC_WRONG;
383  }
384  // Verify CRC_A - do our own calculation and store the control in controlBuffer.
385  byte controlBuffer[2];
386  n = PCD_CalculateCRC(&backData[0], *backLen - 2, &controlBuffer[0]);
387  if (n != STATUS_OK) {
388  return n;
389  }
390  if ((backData[*backLen - 2] != controlBuffer[0]) || (backData[*backLen - 1] != controlBuffer[1])) {
391  return STATUS_CRC_WRONG;
392  }
393  }
394 
395  return STATUS_OK;
396 } // End PCD_CommunicateWithPICC()

487  // Byte 0: SEL Indicates the Cascade Level: PICC_CMD_SEL_CL1, PICC_CMD_SEL_CL2 or PICC_CMD_SEL_CL3
488  // Byte 1: NVB Number of Valid Bits (in complete command, not just the UID): High nibble: complete bytes, Low nibble: Extra bits.
489  // Byte 2: UID-data or CT See explanation below. CT means Cascade Tag.
490  // Byte 3: UID-data
491  // Byte 4: UID-data
492  // Byte 5: UID-data
493  // Byte 6: BCC Block Check Character - XOR of bytes 2-5
494  // Byte 7: CRC_A
495  // Byte 8: CRC_A
496  // The BCC and CRC_A is only transmitted if we know all the UID bits of the current Cascade Level.
497  //
498  // Description of bytes 2-5: (Section 6.5.4 of the ISO/IEC 14443-3 draft: UID contents and cascade levels)
499  // UID size Cascade level Byte2 Byte3 Byte4 Byte5
500  // ======== ============= ===== ===== ===== =====
501  // 4 bytes 1 uid0 uid1 uid2 uid3
502  // 7 bytes 1 CT uid0 uid1 uid2
503  // 2 uid3 uid4 uid5 uid6
504  // 10 bytes 1 CT uid0 uid1 uid2
505  // 2 CT uid3 uid4 uid5
506  // 3 uid6 uid7 uid8 uid9
507 
508  // Sanity checks
509  if (validBits > 80) {
510  return STATUS_INVALID;
511  }
512 
513  // Prepare MFRC522
514  PCD_ClearRegisterBitMask(CollReg, 0x80); // ValuesAfterColl=1 => Bits received after collision are cleared.
515 
516  // Repeat Cascade Level loop until we have a complete UID.
517  uidComplete = false;
518  while ( ! uidComplete) {
519  // Set the Cascade Level in the SEL byte, find out if we need to use the Cascade Tag in byte 2.
520  switch (cascadeLevel) {
521  case 1:
522  buffer[0] = PICC_CMD_SEL_CL1;
523  uidIndex = 0;
524  useCascadeTag = validBits && uid->size > 4; // When we know that the UID has more than 4 bytes
525  break;
526 
527  case 2:
528  buffer[0] = PICC_CMD_SEL_CL2;
529  uidIndex = 3;
530  useCascadeTag = validBits && uid->size > 7; // When we know that the UID has more than 7 bytes
531  break;
532 
533  case 3:
534  buffer[0] = PICC_CMD_SEL_CL3;
535  uidIndex = 6;
536  useCascadeTag = false; // Never used in CL3.
537  break;
538 
539  default:
540  return STATUS_INTERNAL_ERROR;
541  break;
542  }
543 
544  // How many UID bits are known in this Cascade Level?
545  currentLevelKnownBits = validBits - (8 * uidIndex);
546  if (currentLevelKnownBits < 0) {
547  currentLevelKnownBits = 0;
548  }
549  // Copy the known bits from uid->uidByte[] to buffer[]
550  index = 2; // destination index in buffer[]
551  if (useCascadeTag) {
552  buffer[index++] = PICC_CMD_CT;
553  }
554  byte bytesToCopy = currentLevelKnownBits / 8 + (currentLevelKnownBits % 8 ? 1 : 0); // The number of bytes needed to represent the known bits for this level.
555  if (bytesToCopy) {
556  byte maxBytes = useCascadeTag ? 3 : 4; // Max 4 bytes in each Cascade Level. Only 3 left if we use the Cascade Tag
557  if (bytesToCopy > maxBytes) {
558  bytesToCopy = maxBytes;
559  }
560  for (count = 0; count < bytesToCopy; count++) {
561  buffer[index++] = uid->uidByte[uidIndex + count];
562  }
563  }
564  // Now that the data has been copied we need to include the 8 bits in CT in currentLevelKnownBits
565  if (useCascadeTag) {
566  currentLevelKnownBits += 8;
567  }
568 
569  // Repeat anti collision loop until we can transmit all UID bits + BCC and receive a SAK - max 32 iterations.
570  selectDone = false;
571  while ( ! selectDone) {
572  // Find out how many bits and bytes to send and receive.
573  if (currentLevelKnownBits >= 32) { // All UID bits in this Cascade Level are known. This is a SELECT.
574  //Serial.print("SELECT: currentLevelKnownBits="); Serial.println(currentLevelKnownBits, DEC);
575  buffer[1] = 0x70; // NVB - Number of Valid Bits: Seven whole bytes
576  // Calulate BCC - Block Check Character
577  buffer[6] = buffer[2] ^ buffer[3] ^ buffer[4] ^ buffer[5];
578  // Calculate CRC_A
579  result = PCD_CalculateCRC(buffer, 7, &buffer[7]);
580  if (result != STATUS_OK) {
581  return result;
582  }
583  txLastBits = 0; // 0 => All 8 bits are valid.
584  bufferUsed = 9;
585  // Store response in the last 3 bytes of buffer (BCC and CRC_A - not needed after tx)
586  responseBuffer = &buffer[6];
587  responseLength = 3;
588  }
589  else { // This is an ANTICOLLISION.
590  //Serial.print("ANTICOLLISION: currentLevelKnownBits="); Serial.println(currentLevelKnownBits, DEC);
591  txLastBits = currentLevelKnownBits % 8;
592  count = currentLevelKnownBits / 8; // Number of whole bytes in the UID part.
593  index = 2 + count; // Number of whole bytes: SEL + NVB + UIDs
594  buffer[1] = (index << 4) + txLastBits; // NVB - Number of Valid Bits
595  bufferUsed = index + (txLastBits ? 1 : 0);
596  // Store response in the unused part of buffer
597  responseBuffer = &buffer[index];
598  responseLength = sizeof(buffer) - index;
599  }
600 
601  // Set bit adjustments
602  rxAlign = txLastBits; // Having a seperate variable is overkill. But it makes the next line easier to read.
603  PCD_WriteRegister(BitFramingReg, (rxAlign << 4) + txLastBits); // RxAlign = BitFramingReg[6..4]. TxLastBits = BitFramingReg[2..0]
604 
605  // Transmit the buffer and receive the response.
606  result = PCD_TransceiveData(buffer, bufferUsed, responseBuffer, &responseLength, &txLastBits, rxAlign);
607  if (result == STATUS_COLLISION) { // More than one PICC in the field => collision.
608  result = PCD_ReadRegister(CollReg); // CollReg[7..0] bits are: ValuesAfterColl reserved CollPosNotValid CollPos[4:0]
609  if (result & 0x20) { // CollPosNotValid
610  return STATUS_COLLISION; // Without a valid collision position we cannot continue
611  }
612  byte collisionPos = result & 0x1F; // Values 0-31, 0 means bit 32.
613  if (collisionPos == 0) {
614  collisionPos = 32;
615  }
616  if (collisionPos <= currentLevelKnownBits) { // No progress - should not happen
617  return STATUS_INTERNAL_ERROR;
618  }
619  // Choose the PICC with the bit set.
620  currentLevelKnownBits = collisionPos;
621  count = (currentLevelKnownBits - 1) % 8; // The bit to modify
622  index = 1 + (currentLevelKnownBits / 8) + (count ? 1 : 0); // First byte is index 0.
623  buffer[index] |= (1 << count);
624  }
625  else if (result != STATUS_OK) {
626  return result;
627  }
628  else { // STATUS_OK
629  if (currentLevelKnownBits >= 32) { // This was a SELECT.
630  selectDone = true; // No more anticollision
631  // We continue below outside the while.
632  }
633  else { // This was an ANTICOLLISION.
634  // We now have all 32 bits of the UID in this Cascade Level
635  currentLevelKnownBits = 32;
636  // Run loop again to do the SELECT.
637  }
638  }
639  } // End of while ( ! selectDone)
640 
641  // We do not check the CBB - it was constructed by us above.
642 
643  // Copy the found UID bytes from buffer[] to uid->uidByte[]
644  index = (buffer[2] == PICC_CMD_CT) ? 3 : 2; // source index in buffer[]
645  bytesToCopy = (buffer[2] == PICC_CMD_CT) ? 3 : 4;
646  for (count = 0; count < bytesToCopy; count++) {
647  uid->uidByte[uidIndex + count] = buffer[index++];
648  }
649 
650  // Check response SAK (Select Acknowledge)
651  if (responseLength != 3 || txLastBits != 0) { // SAK must be exactly 24 bits (1 byte + CRC_A).
652  return STATUS_ERROR;
653  }
654  // Verify CRC_A - do our own calculation and store the control in buffer[2..3] - those bytes are not needed anymore.
655  result = PCD_CalculateCRC(responseBuffer, 1, &buffer[2]);
656  if (result != STATUS_OK) {
657  return result;
658  }
659  if ((buffer[2] != responseBuffer[1]) || (buffer[3] != responseBuffer[2])) {
660  return STATUS_CRC_WRONG;
661  }
662  if (responseBuffer[0] & 0x04) { // Cascade bit set - UID not complete yes
663  cascadeLevel++;
664  }
665  else {
666  uidComplete = true;
667  uid->sak = responseBuffer[0];
668  }
669  } // End of while ( ! uidComplete)
670 
671  // Set correct uid->size
672  uid->size = 3 * cascadeLevel + 1;
673 
674  return STATUS_OK;
675 } // End PICC_Select()

1260  // g[3] Access bits for the sector trailer, block 3 (for sectors 0-31) or block 15 (for sectors 32-39)
1261  // g[2] Access bits for block 2 (for sectors 0-31) or blocks 10-14 (for sectors 32-39)

1270  byte group; // 0-3 - active group for access bits
1271  bool firstInGroup; // True for the first block dumped in the group

1278  else if (sector < 40) { // Sectors 32-39 has 16 blocks each
1279  no_of_blocks = 16;
1280  firstBlock = 128 + (sector - 32) * no_of_blocks;
1281  }
1282  else { // Illegal input, no MIFARE Classic PICC has more than 40 sectors.

1284  }

1287  byte byteCount;
1288  byte buffer[18];

1296  Serial.print(sector);
1297  Serial.print(" ");
1298  }
1299  else {
1300  Serial.print(" ");
1301  }
1302  // Block number
1303  Serial.print(blockAddr < 10 ? " " : (blockAddr < 100 ? " " : " ")); // Pad with spaces
1304  Serial.print(blockAddr);
1305  Serial.print(" ");

1307  if (isSectorTrailer) {
1308  status = PCD_Authenticate(PICC_CMD_MF_AUTH_KEY_A, firstBlock, key, uid);
1309  if (status != STATUS_OK) {
1310  Serial.print("PCD_Authenticate() failed: ");
1311  Serial.println(GetStatusCodeName(status));
1312  return;
1313  }
1314  }

1316  byteCount = sizeof(buffer);
1317  status = MIFARE_Read(blockAddr, buffer, &byteCount);

1320  Serial.println(GetStatusCodeName(status));
1321  continue;
1322  }

1325  Serial.print(buffer[index] < 0x10 ? " 0" : " ");
1326  Serial.print(buffer[index], HEX);
1327  if ((index % 4) == 3) {
1328  Serial.print(" ");
1329  }
1330  }

1332  if (isSectorTrailer) {
1333  c1 = buffer[7] >> 4;
1334  c2 = buffer[8] & 0xF;
1335  c3 = buffer[8] >> 4;
1336  c1_ = buffer[6] & 0xF;
1337  c2_ = buffer[6] >> 4;
1338  c3_ = buffer[7] & 0xF;
1339  invertedError = (c1 != (~c1_ & 0xF)) || (c2 != (~c2_ & 0xF)) || (c3 != (~c3_ & 0xF));
1340  g[0] = ((c1 & 1) << 2) | ((c2 & 1) << 1) | ((c3 & 1) << 0);
1341  g[1] = ((c1 & 2) << 1) | ((c2 & 2) << 0) | ((c3 & 2) >> 1);
1342  g[2] = ((c1 & 4) << 0) | ((c2 & 4) >> 1) | ((c3 & 4) >> 2);
1343  g[3] = ((c1 & 8) >> 1) | ((c2 & 8) >> 2) | ((c3 & 8) >> 3);
1344  isSectorTrailer = false;
1345  }

1348  if (no_of_blocks == 4) {
1349  group = blockOffset;
1350  firstInGroup = true;
1351  }
1352  else {

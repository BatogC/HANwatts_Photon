29 #include "application.h"
30 #include "MQTT.h"
31 
32 void callback(char* topic, byte* payload, unsigned int length);
33 MQTT client("iot.eclipse.org", 1883, callback);
34 
35 // recieve message
36 void callback(char* topic, byte* payload, unsigned int length) {
37  char p[length + 1];
38  memcpy(p, payload, length);
39  p[length] = NULL;
40 
41  if (!strcmp(p, "RED"))
42  RGB.color(255, 0, 0);
43  else if (!strcmp(p, "GREEN"))
44  RGB.color(0, 255, 0);
45  else if (!strcmp(p, "BLUE"))
46  RGB.color(0, 0, 255);
47  else
48  RGB.color(255, 255, 255);
49  delay(1000);
50 }
51 
52 
53 void setup() {
54  RGB.control(true);
55 
56  // connect to the server(unique id by Time.now())
57  client.connect("sparkclient_" + String(Time.now()));
58 
59  // publish/subscribe
60  if (client.isConnected()) {
61  client.publish("outTopic/message","hello world");
62  client.subscribe("inTopic/message");
63  }
64 }
65

69 }

71 ## FAQ
72 ### Can't connect/publish/subscribe to the MQTT server?
73 - Check your MQTT server and port(default 1883) is really working with the mosquitto_pub/sub command. And maybe your MQTT server can't connect from Internet because of firewall. Check your network environments.
74 - Check your subscribe/publish topic name is really matched.
75 - Perhaps device firmware network stack is failed. check your firmware version and bugs.
76 - If you use MQTT-TLS, check your RooT CA pem file, client key, certifications is okay or not.
77 - Several MQTT server will disconnect to the 1st connection when you use the same user_id. When the application call the connect method, use different user_id in every devices in connect method's 2nd argument. Use MAC address as a user_id will be better.
78 <pre>
79  // device.1
80  client.connect("spark-client", "user_1", "password1");
81  // other devices...
82  client.connect("spark-client", "user_others", "password1");
83 </pre>
84 
85 ### I want to change MQTT keep alive timeout.
86 MQTT keepalive timeout is defined "MQTT_DEFAULT_KEEPALIVE 15"(15 sec) in header file. You can change the keepalive timeout in constructor.
87 <pre>
88  MQTT client("server_name", 1883, callback); // default: send keepalive packet to MQTT server in every 15sec.
89  MQTT client("server_name", 1883, 30, callback); // keepliave timeout is 30sec.
90 </pre>
91 
92 ### Want to use over the 255 message size.
93 In this library, max MQTT message size is defined "MQTT_MAX_PACKET_SIZE 255" in header file. But If you want to use over 255bytes, use the constructor 4th argument.
94 <pre>
95  MQTT client("server_name", 1883, callback); // default 255bytes
96  MQTT client("server_name", 1883, MQTT_DEFAULT_KEEPALIVE, callback, 512); // max 512bytes
97 </pre>
98 
99 ### Can I use on old firmware?
100 No, use default latest firmware. I test this library on default latest firmware or latest pre-release version. If you really want to use old firmware(I think don't need that case), maybe it can't work well and it is out of my assumption.

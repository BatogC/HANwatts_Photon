182 size_t Print::printNumber(unsigned long n, uint8_t base) {
183  char buf[8 * sizeof(long) + 1]; // Assumes 8-bit chars plus zero byte.
184  char *str = &buf[sizeof(buf) - 1];
185 
186  *str = '\0';
187 
188  // prevent crash if called with base == 1
189  if (base < 2) base = 10;
190 
191  do {
192  unsigned long m = n;
193  n /= base;
194  char c = m - base * n;
195  *--str = c < 10 ? c + '0' : c + 'A' - 10;
196  } while(n);
197 
198  return write(str);
199 }

201 size_t Print::printFloat(double number, uint8_t digits)
202 {
203  size_t n = 0;
204 
205  if (isnan(number)) return print("nan");
206  if (isinf(number)) return print("inf");
207  if (number > 4294967040.0) return print ("ovf"); // constant determined empirically
208  if (number <-4294967040.0) return print ("ovf"); // constant determined empirically
209 
210  // Handle negative numbers
211  if (number < 0.0)
212  {
213  n += print('-');
214  number = -number;
215  }
216 
217  // Round correctly so that print(1.999, 2) prints as "2.00"
218  double rounding = 0.5;
219  for (uint8_t i=0; i<digits; ++i)
220  rounding /= 10.0;
221 
222  number += rounding;
223 
224  // Extract the integer part of the number and print it
225  unsigned long int_part = (unsigned long)number;
226  double remainder = number - (double)int_part;
227  n += print(int_part);
228 
229  // Print the decimal point, but only if there are digits beyond
230  if (digits > 0) {
231  n += print(".");
232  }
233 
234  // Extract digits from the remainder one at a time
235  while (digits-- > 0)
236  {
237  remainder *= 10.0;
238  int toPrint = int(remainder);
239  n += print(toPrint);
240  remainder -= toPrint;
241  }
242 
243  return n;
244 }

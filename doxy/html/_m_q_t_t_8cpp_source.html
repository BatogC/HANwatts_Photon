195 uint16_t MQTT::readPacket(uint8_t* lengthLength) {
196  uint16_t len = 0;
197  buffer[len++] = readByte();
198  bool isPublish = (buffer[0]&0xF0) == MQTTPUBLISH;
199  uint32_t multiplier = 1;
200  uint16_t length = 0;
201  uint8_t digit = 0;
202  uint16_t skip = 0;
203  uint8_t start = 0;
204 
205  do {
206  digit = readByte();
207  buffer[len++] = digit;
208  length += (digit & 127) * multiplier;
209  multiplier *= 128;
210  } while ((digit & 128) != 0);
211  *lengthLength = len-1;
212 
213  if (isPublish) {
214  // Read in topic length to calculate bytes to skip over for Stream writing
215  buffer[len++] = readByte();
216  buffer[len++] = readByte();
217  skip = (buffer[*lengthLength+1]<<8)+buffer[*lengthLength+2];
218  start = 2;
219  if (buffer[0] & MQTTQOS1_HEADER_MASK) {
220  // skip message id
221  skip += 2;
222  }
223  }
224 
225  for (uint16_t i = start;i<length;i++) {
226  digit = readByte();
227  if (len < this->maxpacketsize) {
228  buffer[len] = digit;
229  }
230  len++;
231  }
232 
233  if (len > this->maxpacketsize) {
234  len = 0; // This will cause the packet to be ignored.
235  }
236 
237  return len;
